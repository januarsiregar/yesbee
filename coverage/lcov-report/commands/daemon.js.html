// jshint esnext: true
 
const bono = require('bono');
const url = require('url');
const fs = require('fs-promise');
const path = require('path');
const _ = require('lodash');
 
module.exports = (function() {
  'use strict';
 
  function *prepareComponents (execution) {
    var componentFiles;
 
    // core components
    var baseDir = path.resolve(__dirname, '../components');
    try {
      componentFiles = yield fs.readdir(baseDir);
    } catch(e) {
      componentFiles = [];
    }
    _.forEach(componentFiles, function(componentFile) {
      if (path.extname(componentFile) === '.js') {
        var name = path.basename(componentFile, '.js');
        execution.app.addComponent(name, require(path.join(baseDir, name)));
      }
    });
 
    // container-specific components
    baseDir = path.resolve('./components');
    try {
      componentFiles = yield fs.readdir(baseDir);
    } catch(e) {
      componentFiles = [];
    }
    _.forEach(componentFiles, function(componentFile) {
      if (path.extname(componentFile) === '.js') {
        var name = path.basename(componentFile, '.js');
        execution.app.addComponent(name, require(path.join(baseDir, name)));
      }
    });
  }
 
  function *prepareServices (execution) {
    var serviceFiles;
 
    // core services
    var baseDir = path.resolve('./services');
    try {
      serviceFiles = yield fs.readdir(baseDir);
    } catch(e) {
      serviceFiles = [];
    }
    _.forEach(serviceFiles, function(serviceFile) {
      if (path.extname(serviceFile) === '.js') {
        var name = path.basename(serviceFile, '.js');
        execution.app.addService(name, require(path.join(baseDir, name)));
      }
    });
  }
 
  function *prepareDaemon (execution) {
    execution.daemon = bono();
    execution.app.callback = function() {
      return execution.daemon.callback();
    };
 
    execution.daemon.addMiddleware(function *(next) {
      execution.logger({$name:'daemon', message: this.method + ' ' + this.url});
 
      var segments = this.url.split('/');
 
      try {
        var result = yield execution.invoke({
          id: segments[1],
          args: segments.slice(2),
          execution: this,
        });
        this.status = 200;
        this.body = result;
      } catch(e) {
        if (e.message.indexOf('Cannot find module') !== 0) {
          this.status = 500;
          this.body = {
            error: e.message
          };
        }
      }
    });
 
    if (execution.app.host) {
      var parsed = url.parse(execution.app.host);
      execution.daemon.listen(parsed.port, parsed.hostname, function() {
        execution.logger({message: 'Listening at ' + execution.app.host });
      });
    } else if (execution.app.socketPath) {
      try {
        fs.unlinkSync(execution.app.socketPath);
      } catch(e) {}
 
      yield new Promise(function(resolve, reject) {
        execution.daemon.listen(execution.app.socketPath, function() {
          execution.logger({message: 'Listening at ' + execution.app.socketPath });
          resolve();
        });
      });
    } else {
      throw new Error('Cannot bind daemon to either host or socket path');
    }
  }
 
  function *daemonCommand (execution) {
    yield prepareComponents(execution);
    yield prepareServices(execution);
    yield prepareDaemon(execution);
  }
  return daemonCommand;
})();
